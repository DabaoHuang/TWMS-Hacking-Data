[ENABLE]
alloc(GetHPMP,128)
label(MM)
label(NM)
label(MH)
label(NH)
alloc(MAXMP,4)
alloc(NOWMP,4)
alloc(MAXHP,4)
alloc(NOWHP,4)
registersymbol(MAXMP)
registersymbol(NOWMP)
registersymbol(MAXHP)
registersymbol(NOWHP)

GetHPMP:
call 008B89B3
pop ebp

cmp [esp],0074FF43
je MM
cmp [esp],0074FF51
je NM
cmp [esp],0074FF66
je MH
cmp [esp],0074FF74
je NH
ret

MM:
mov [MAXMP],eax
ret

NM:
mov [NOWMP],eax
ret

MH:
mov [MAXHP],eax
ret

NH:
mov [NOWHP],eax
ret

00404DA1:
jmp GetHPMP
[DISABLE]
00404DA1:
call 008B89B3
dealloc(GetHPMP)
dealloc(MAXMP)
dealloc(NOWMP)
dealloc(MAXHP)
dealloc(NOWHP)
unregistersymbol(MAXMP)
unregistersymbol(NOWMP)
unregistersymbol(MAXHP)
unregistersymbol(NOWHP)
