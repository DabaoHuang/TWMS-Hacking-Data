//------------------------------------------------------------------

jmp 0085FD2B+5
//------------------------------------------------------------------
AttackTimes:

//------------------------------------------------------------------
0085FD2B://無延遲 // E8 ?? ?? ?? 00 83 C4 0C E8 ?? ?? ?? 00 C2 04 00
